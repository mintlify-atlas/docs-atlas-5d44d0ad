<Note>
  DataTable is a feature-rich table component built on TanStack Table that provides sorting, filtering, pagination, row selection, column resizing, and more.
</Note>

## Import

```svelte
import { DataTable } from 'popui'
```

## Basic Usage

```svelte
<script lang="ts">
  import { DataTable } from 'popui'
  
  type User = {
    id: string
    name: string
    email: string
    status: string
  }
  
  const data: User[] = [
    { id: '1', name: 'John Doe', email: 'john@example.com', status: 'active' },
    { id: '2', name: 'Jane Smith', email: 'jane@example.com', status: 'inactive' }
  ]
  
  const columns = [
    {
      id: 'name',
      accessorKey: 'name',
      header: 'Name',
      cellType: 'text'
    },
    {
      id: 'email',
      accessorKey: 'email',
      header: 'Email',
      cellType: 'text'
    },
    {
      id: 'status',
      accessorKey: 'status',
      header: 'Status',
      cellType: 'tag',
      cellConfig: {
        options: [
          { value: 'active', label: 'Active', color: 'green' },
          { value: 'inactive', label: 'Inactive', color: 'grey' }
        ]
      }
    }
  ]
</script>

<DataTable {data} {columns} />
```

## Column Types

DataTable supports multiple built-in cell types:

### Text Cell

```svelte
{
  id: 'name',
  accessorKey: 'name',
  header: 'Name',
  cellType: 'text',
  cellConfig: {
    className: 'font-bold'
  }
}
```

### Boolean Cell

Display icons based on boolean values:

```svelte
<script>
  import { Check } from '@invopop/ui-icons'
</script>

{
  id: 'verified',
  accessorKey: 'verified',
  header: 'Verified',
  cellType: 'boolean',
  cellConfig: {
    icon: Check,
    iconClass: 'size-4 text-green-500',
    showWhenTrue: true,
    showWhenFalse: false,
    hintWhenTrue: 'Verified user',
    hintWhenFalse: 'Not verified'
  }
}
```

### Tag Cell

Display status tags with colors:

```svelte
{
  id: 'status',
  accessorKey: 'status',
  header: 'Status',
  cellType: 'tag',
  cellConfig: {
    options: [
      { value: 'active', label: 'Active', color: 'green' },
      { value: 'pending', label: 'Pending', color: 'yellow' },
      { value: 'inactive', label: 'Inactive', color: 'grey' }
    ],
    showDot: true
  }
}
```

### Date Cell

## Custom Cell Renderer

You can provide custom cell renderers using a function or Snippet:

```svelte
<script lang="ts">
  import { DataTable } from 'popui'
  import CustomComponent from './CustomComponent.svelte'
  import { renderComponent } from 'popui/data-table'
  
  const columns = [
    {
      id: 'custom',
      accessorKey: 'name',
      header: 'Custom',
      cell: (value, row) => {
        return renderComponent(CustomComponent, { value, data: row })
      }
    }
  ]
</script>
```

## Sorting

Enable sorting on columns:

```svelte
<script lang="ts">
  const columns = [
    {
      id: 'name',
      accessorKey: 'name',
      header: 'Name',
      cellType: 'text',
      enableSorting: true // Enable sorting (default: true)
    }
  ]
</script>

<DataTable 
  {data} 
  {columns}
  initialSortColumn="name"
  initialSortDirection="asc"
/>
```

## Manual Sorting (Server-Side)

```svelte
<script lang="ts">
  function handleSortingChange(columnId: string, direction: 'asc' | 'desc' | '') {
    // Fetch sorted data from server
    console.log('Sort by:', columnId, direction)
  }
</script>

<DataTable 
  {data} 
  {columns}
  manualSorting={true}
  onSortingChange={handleSortingChange}
/>
```

## Manual Pagination (Server-Side)

```svelte
<script lang="ts">
  let currentPage = 0
  let pageSize = 10
  let totalPages = 10
  let totalRows = 100
  
  function handlePageChange(pageIndex: number) {
    currentPage = pageIndex
    // Fetch data for new page
  }
  
  function handlePageSizeChange(newPageSize: number) {
    pageSize = newPageSize
    currentPage = 0
    // Fetch data with new page size
  }
</script>

<DataTable 
  {data} 
  {columns}
  manualPagination={true}
  pageCount={totalPages}
  rowCount={totalRows}
  initialPage={currentPage}
  initialPageSize={pageSize}
  onPageChange={handlePageChange}
  onPageSizeChange={handlePageSizeChange}
/>
```

## Row Actions

```svelte
<script lang="ts">
  import { Edit, Trash } from '@invopop/ui-icons'
  
  const rowActions = [
    { label: 'Edit', value: 'edit', icon: Edit },
    { label: 'Delete', value: 'delete', icon: Trash, destructive: true }
  ]
  
  function handleRowAction(action, row) {
    console.log('Action:', action, 'Row:', row)
  }
</script>

<DataTable 
  {data} 
  {columns}
  {rowActions}
  onRowAction={handleRowAction}
/>
```

### Dynamic Row Actions

```svelte
<script lang="ts">
  function getRowActions(row) {
    if (row.status === 'active') {
      return [
        { label: 'Deactivate', value: 'deactivate' },
        { label: 'Edit', value: 'edit' }
      ]
    }
    return [
      { label: 'Activate', value: 'activate' }
    ]
  }
</script>

<DataTable 
  {data} 
  {columns}
  {getRowActions}
  onRowAction={handleRowAction}
/>
```

## Column Management

### Column Visibility

```svelte
<DataTable 
  {data} 
  {columns}
  initialColumnVisibility={{
    email: false // Hide email column initially
  }}
  onColumnVisibilityChange={(visibility) => {
    console.log('Visibility changed:', visibility)
  }}
/>
```

### Column Resizing

```svelte
<script lang="ts">
  const columns = [
    {
      id: 'name',
      accessorKey: 'name',
      header: 'Name',
      cellType: 'text',
      size: 200,
      minSize: 100,
      maxSize: 400,
      enableResizing: true
    }
  ]
</script>

<DataTable 
  {data} 
  {columns}
  onColumnResize={(sizes) => {
    console.log('Column sizes:', sizes)
  }}
/>
```

### Frozen Columns

```svelte
<DataTable 
  {data} 
  {columns}
  initialFrozenColumns={['name', 'email']}
  onFreezeChange={(columnId) => {
    console.log('Freeze toggled for:', columnId)
  }}
/>
```

### Column Ordering

```svelte
<DataTable

  {columns}

```

## Row Styling

```svelte
<script lang="ts">
  function getRowClassName(row) {
    return row.status === 'critical' ? 'bg-red-50' : ''
  }
  
  function getRowState(row) {
    return {
      isError: row.status === 'error',
      isWarning: row.status === 'warning',
      isSuccess: row.status === 'success'
    }
  }
</script>

<DataTable 
  {data} 
  {columns}
  {getRowClassName}
  {getRowState}
/>
```

## Keyboard Navigation

By default, DataTable supports keyboard navigation:

- **Arrow Up/Down**: Navigate rows
- **Shift + Arrow Up/Down**: Select multiple rows
- **Space/Enter**: Select current row
- **Escape**: Clear focus

To disable keyboard navigation:

```svelte
<DataTable {data} {columns} disableKeyboardNavigation={true} />
```

## Column Definition

<ParamField path="id" type="string" required>
  Unique identifier for the column.
</ParamField>

<ParamField path="accessorKey" type="keyof TData">
  Key to access data in the row object.
</ParamField>

<ParamField path="header" type="string">
  Header text to display. Use empty string or omit for no header.
</ParamField>

<ParamField path="cellType" type="'text' | 'boolean' | 'tag' | 'date' | 'currency' | 'uuid' | 'custom'">
  Built-in cell renderer type.
</ParamField>

<ParamField path="cellConfig" type="CellConfig">
  Configuration object for the cell renderer (varies by cellType).
</ParamField>

<ParamField path="cell" type="Snippet | Function">
  Custom cell renderer function or Snippet.
</ParamField>

<ParamField path="enableSorting" type="boolean" default="true">
  Enable sorting for this column.
</ParamField>

<ParamField path="enableHiding" type="boolean" default="true">
  Allow this column to be hidden.
</ParamField>

<ParamField path="enableResizing" type="boolean" default="true">
  Allow this column to be resized.
</ParamField>

<ParamField path="disableColumnFilter" type="boolean" default="false">
  Disable column filter option.
</ParamField>

<ParamField path="size" type="number">
  Default column width in pixels.
</ParamField>

<ParamField path="minSize" type="number">
  Minimum column width in pixels.
</ParamField>

<ParamField path="maxSize" type="number">
  Maximum column width in pixels.
</ParamField>

<ParamField path="loadingConfig" type="LoadingConfig">
  Configuration for loading skeleton placeholders.
</ParamField>
